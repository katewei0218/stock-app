// 計算每人在每帳戶的可用資金（方案B）

function calcFundsData(){

  // result[accId][person] = {in, out, buyOut, sellIn}

  const result={};

  S.accounts.forEach(a=>{

    result[a.id]={};

    S.persons.forEach(p=>result[a.id][p]={in:0,out:0,buyOut:0,sellIn:0});

  });

  // 資金流入/出

  S.funds.forEach(f=>{

    if(!result[f.acc])return;

    if(!result[f.acc][f.person])result[f.acc][f.person]={in:0,out:0,buyOut:0,sellIn:0};

    if(f.type==='in')result[f.acc][f.person].in+=f.amount;

    else result[f.acc][f.person].out+=f.amount;

  });

  // 買入出資（按每人出資%）

  S.buys.filter(b=>!b.isStockDiv).forEach(b=>{

    if(!result[b.acc])return;

    b.investors.forEach(inv=>{

      if(!result[b.acc][inv.person])result[b.acc][inv.person]={in:0,out:0,buyOut:0,sellIn:0};

      result[b.acc][inv.person].buyOut+=b.netCost*(inv.pct/100);

    });

  });

  // 賣出回收（按每人分回金額）

  S.sells.forEach(s=>{

    if(!result[s.acc])return;

    Object.entries(s.personPnl||{}).forEach(([person,v])=>{

      if(!result[s.acc][person])result[s.acc][person]={in:0,out:0,buyOut:0,sellIn:0};

      result[s.acc][person].sellIn+=v.income;

    });

  });

  return result;

}

function getHoldings(accId){

function renderDividends(){

  const yr=document.getElementById('dv-yr')?.value||'';

  const accF=document.getElementById('dv-acc')?.value||'';

  const list=[...S.divs].filter(d=>(!yr||d.date.split('-')[0]===yr)&&(!accF||d.acc===accF)).sort((a,b)=>b.date.localeCompare(a.date));

  const totalCash=list.filter(d=>d.type==='cash').reduce((s,d)=>s+d.total,0);

  const totalStock=list.filter(d=>d.type==='stock').reduce((s,d)=>s+d.shares,0);

  document.getElementById('dv-summary').innerHTML=`

    <div class="sg">

      <div class="sb"><div class="sl">現金股息合計</div><div class="sv pos">${fmtM(totalCash)}</div></div>

      <div class="sb"><div class="sl">配股合計</div><div class="sv" style="color:var(--blue)">${totalStock.toLocaleString()} 股</div></div>

    </div>`;


  // 各成員配息估算（依流入佔比）

  const fd=calcFundsData();

  const personDiv={};

  list.filter(d=>d.type==='cash').forEach(d=>{

    const ap=fd[d.acc]||{};

    const totalIn=Object.values(ap).reduce((s,v)=>s+v.in,0);

    if(!totalIn){if(!personDiv['未分配'])personDiv['未分配']=0;personDiv['未分配']+=d.total;return;}

    Object.entries(ap).forEach(([person,v])=>{

      if(!personDiv[person])personDiv[person]=0;

      personDiv[person]+=d.total*(v.in/totalIn);

    });

  });

  const personDivHtml=Object.keys(personDiv).length

    ?`<div class="card"><div class="ct">各成員配息估算（依流入佔比）</div>`+

      Object.entries(personDiv).map(([p,v])=>`

        <div style="display:flex;justify-content:space-between;align-items:center;padding:10px 0;border-bottom:1px solid var(--border)">

          <span style="font-weight:600">👤 ${p}</span><span style="font-weight:700" class="pos">＋${fmtM(v)}</span>

        </div>`).join('')+'</div>':'' ;


  document.getElementById('div-list').innerHTML=personDivHtml+(list.length?list.map(d=>{

    const accName=S.accounts.find(a=>a.id===d.acc)?.name||d.acc;

    const isCash=d.type==='cash';

    return `<div class="li">

      <div class="lh">

        <div><span class="tag ${isCash?'td':'tsd'}">${isCash?'現金股息':'股票股利'}</span>

          <span style="font-weight:700;font-size:14px;margin-left:8px">${d.name}</span>

          <span style="font-size:11px;color:var(--muted);margin-left:4px">${d.code}</span></div>

        <button class="db2" onclick="deleteDiv('${d.id}')">✕</button>

      </div>

      <div style="font-size:12px;color:var(--text2);margin-bottom:6px">📅 ${d.date} · 🏦 ${accName}</div>

      ${isCash

        ?`<div style="display:flex;gap:10px;flex-wrap:wrap">

            <span style="font-size:12px">每股 <b>${d.perShare}</b> 元</span>

            <span style="font-size:12px">股數 <b>${d.shares.toLocaleString()}</b></span>

            <span style="font-size:13px;font-weight:700" class="pos">＋${fmt(d.total)} 元</span>

          </div>`

        :`<div style="font-size:13px;font-weight:700;color:var(--blue)">配股 ＋${d.shares.toLocaleString()} 股</div>`}

      ${d.note?`<div style="font-size:11px;color:var(--muted);margin-top:4px">${d.note}</div>`:''}

    </div>`;

  }).join(''):'<div class="empty">尚無股息紀錄</div>');

}
